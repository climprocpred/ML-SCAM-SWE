 44class DiscoMlpTransformer(nn.Module):
 45    """
 46    Two-branch model:
 47      - MLP encoder on per-location channels
 48      - DiSCO convolution branch
 49    Branches are concatenated (no fusion MLP), with a learnable scalar gate
 50    applied to DiSCO (conservative init).
 51    
 52    Output is processed by a stack of SphericalAttentionBlocks (neighborhood attention).
 53    """
 54
 55    def __init__(self, cfg: DiscoMlpTransformerConfig):
 56        super().__init__()
 57        self.cfg = cfg
 58
 59        self.in_shape = cfg.in_shape
 60        self.out_shape = cfg.out_shape
 61
 62        if cfg.mlp_hidden_dim is None:
 63            cfg.mlp_hidden_dim = cfg.mlp_out_dim
 64        
 65        self.mlp_out_dim = cfg.mlp_out_dim
 66        self.disco_out_dim = cfg.disco_out_dim
 67
 68        # MLP branch (per-location)
 69        self.mlp = nn.Sequential(
 70            nn.Linear(cfg.in_channels, cfg.mlp_hidden_dim),
 71            nn.GELU(),
 72            nn.Linear(cfg.mlp_hidden_dim, self.mlp_out_dim),
 73        )
 74
 75        # DiSCO branch
 76        self.disco = DiscreteContinuousConvS2(
 77            in_channels=cfg.in_channels,
 78            out_channels=self.disco_out_dim,
 79            in_shape=cfg.in_shape,
 80            out_shape=cfg.out_shape,
 81            kernel_shape=cfg.kernel_shape,
 82            basis_type=cfg.basis_type,
 83            basis_norm_mode=cfg.basis_norm_mode,
 84            groups=cfg.groups,
 85            grid_in=cfg.grid_in,
 86            grid_out=cfg.grid_out,
 87            bias=cfg.bias,
 88            theta_cutoff=cfg.disco_theta_cutoff,
 89        )
 90
 91        # Conservative DiSCO gate
 92        self.disco_gate = nn.Parameter(torch.tensor(cfg.disco_gate_init))
 93
 94        # Lat sin/cos cache (for out_shape) - only used if add_lat_sincos is True
 95        if cfg.add_lat_sincos:
 96            nlat_out, nlon_out = cfg.out_shape
 97            lat = torch.linspace(-0.5 * torch.pi, 0.5 * torch.pi, nlat_out)
 98            self.register_buffer("_lat_sin", torch.sin(lat).view(1, nlat_out, 1, 1), persistent=False)
 99            self.register_buffer("_lat_cos", torch.cos(lat).view(1, nlat_out, 1, 1), persistent=False)
100
101        # --- Spherical Transformer Setup ---
102        
103        extra_lat = 2 if cfg.add_lat_sincos else 0
104        d_model = self.mlp_out_dim + self.disco_out_dim + extra_lat
105        nlat_t, nlon_t = cfg.out_shape
106
107        # 1. Positional Embedding
108        if cfg.pos_embed == "sequence":
109            self.pos_embed = SequencePositionEmbedding((nlat_t, nlon_t), num_chans=d_model, grid=cfg.grid_internal_pos)
110        elif cfg.pos_embed == "spectral":
111            self.pos_embed = SpectralPositionEmbedding((nlat_t, nlon_t), num_chans=d_model, grid=cfg.grid_internal_pos)
112        elif cfg.pos_embed == "learnable lat":
113            self.pos_embed = LearnablePositionEmbedding((nlat_t, nlon_t), num_chans=d_model, grid=cfg.grid_internal_pos, embed_type="lat")
114        elif cfg.pos_embed == "learnable latlon":
115            self.pos_embed = LearnablePositionEmbedding((nlat_t, nlon_t), num_chans=d_model, grid=cfg.grid_internal_pos, embed_type="latlon")
116        elif cfg.pos_embed == "none":
117            self.pos_embed = nn.Identity()
118        else:
119            raise ValueError(f"Unknown position embedding type {cfg.pos_embed}")
120
121        # 2. Stack of Spherical Attention Blocks
122        self.transformer = nn.Sequential(*[
123            SphericalAttentionBlock(
124                in_shape=cfg.out_shape,
125                out_shape=cfg.out_shape,
126                grid_in=cfg.grid_out,    # Input is on output grid of disco
127                grid_out=cfg.grid_out,   # Output stays on same grid
128                in_chans=d_model,
129                out_chans=d_model,
130                num_heads=cfg.nhead,
131                mlp_ratio=cfg.mlp_ratio,
132                drop_rate=cfg.dropout,
133                drop_path=0.0, # Could expose drop_path_rate if needed
134                act_layer=nn.GELU,
135                norm_layer=cfg.norm_layer,
136                use_mlp=True,
137                bias=cfg.bias,
138                attention_mode=cfg.attention_mode,
139                theta_cutoff=cfg.attn_theta_cutoff,
140            ) for _ in range(cfg.num_layers)
141        ])
142
143        # Decoder MLP
144        # Matches Encoder: Linear(d_model_core -> hidden) -> GELU -> Linear(hidden -> in_channels)
145        extra_lat = 2 if cfg.add_lat_sincos else 0
146        d_model_core = self.mlp_out_dim + self.disco_out_dim + extra_lat
147        self.decoder_mlp = nn.Sequential(
148            nn.Linear(d_model_core, cfg.mlp_hidden_dim),
149            nn.GELU(),
150            nn.Linear(cfg.mlp_hidden_dim, cfg.in_channels),
151        )
152
153    def _lat_sincos(self, device: torch.device, dtype: torch.dtype, nlon: int) -> torch.Tensor:
154        lat_sin = self._lat_sin.to(device=device, dtype=dtype).expand(1, -1, nlon, 1)
155        lat_cos = self._lat_cos.to(device=device, dtype=dtype).expand(1, -1, nlon, 1)
156        return torch.cat([lat_sin, lat_cos], dim=-1)
157
158    def forward(self, x: torch.Tensor) -> torch.Tensor:
159        """
160        x: [B, C, H, W]
161        returns: [B, F, H, W] where F = 2*embed_dim (+2 if add_lat_sincos)
162        """
163        b, c, h, w = x.shape
164        if (h, w) != self.out_shape:
165            raise ValueError(
166                f"Input spatial shape {(h, w)} must match out_shape {self.out_shape} "
167                "for branch concat without resampling."
168            )
169
170        # MLP branch (per-location)
171        x_mlp = x.permute(0, 2, 3, 1)  # [B, H, W, C]
172        x_mlp = self.mlp(x_mlp)        # [B, H, W, E]
173
174        # DiSCO branch
175        x_disco = self.disco(x)        # [B, E, H, W]
176        x_disco = (self.disco_gate * x_disco).permute(0, 2, 3, 1)  # [B, H, W, E]
177
178        # Concatenate branches
179        x_cat = torch.cat([x_mlp, x_disco], dim=-1)  # [B, H, W, 2E]
180
181        # Append latitude sin/cos
182        if self.cfg.add_lat_sincos:
183            lat = self._lat_sincos(device=x.device, dtype=x.dtype, nlon=w)  # [1, H, W, 2]
184            x_cat = torch.cat([x_cat, lat.expand(b, -1, -1, -1)], dim=-1) # [B, H, W, 2E+2]
185        
186        # Prepare for Spherical Transformer: Needs [B, C, H, W]
187        out = x_cat.permute(0, 3, 1, 2).contiguous() # [B, F, H, W]
188
189        # Apply Positional Embedding
190        if isinstance(self.pos_embed, nn.Identity):
191             pass # Skip
192        else:
193             # Pos embed typically expects [B, C, H, W]
194             out = self.pos_embed(out)
195        
196        # Pass through Transformer Stack
197        out = self.transformer(out) # [B, F, H, W]
198
199        # Decoder Strategy:
200        # 1. Permute to [B, H, W, F] for Linear layer
201        out = out.permute(0, 2, 3, 1)
202
203        # 2. Strip the last 'extra_lat' channels if they were added
204        # if self.cfg.add_lat_sincos:
205        #      lat_dims = 2
206        #      out = out[..., :-lat_dims] # [B, H, W, d_model_core]
207
208        # 3. Project back to input channels
209        out = self.decoder_mlp(out) # [B, H, W, in_channels]
210        
211        # 4. Permute back to [B, C, H, W]
212        out = out.permute(0, 3, 1, 2)
213
214        return out
