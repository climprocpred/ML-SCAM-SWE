 57    def set_gridded_ics(self, ics, shuffle=False):
 58        """Set user-provided gridded initial conditions.
 59        
 60        Parameters
 61        ----------
 62        ics : torch.Tensor or list of torch.Tensor
 63            Initial conditions in grid space.
 64            - If tensor: shape (N, 3, nlat, nlon) for N initial conditions
 65            - If list: each element should be (3, nlat, nlon)
 66        shuffle : bool, optional
 67            If True, shuffle the ICs randomly. Default False (sequential).
 68        """
 69        # Convert list to stacked tensor if needed
 70        if isinstance(ics, list):
 71            ics = torch.stack(ics, dim=0)
 72        
 73        # Validate shape
 74        if ics.dim() != 4:
 75            raise ValueError(f"Expected 4D tensor (N, C, H, W), got shape {ics.shape}")
 76        
 77        n_ics, n_chans, nlat, nlon = ics.shape
 78        if nlat != self.nlat or nlon != self.nlon:
 79            raise ValueError(
 80                f"IC grid dimensions ({nlat}, {nlon}) don't match dataset ({self.nlat}, {self.nlon})"
 81            )
 82        
 83        # Convert all ICs from grid space to spectral space
 84        ics = ics.to(self.device)
 85        spectral_ics = []
 86        for i in range(n_ics):
 87            spec_ic = self.solver.grid2spec(ics[i])
 88            spectral_ics.append(spec_ic)
 89        
 90        self._gridded_ics = torch.stack(spectral_ics, dim=0)
 91        
 92        if shuffle:
 93            perm = torch.randperm(len(self._gridded_ics))
 94            self._gridded_ics = self._gridded_ics[perm]
 95        
 96        self._gridded_ic_index = 0
 97        self.ictype = "gridded"
 98        self.num_examples = n_ics
 99        
100        print(f"Set {n_ics} gridded initial conditions")

257    def _get_initial_spec(self):
258        """Get initial condition in spectral space based on current mode.
259        
260        If ic_spinup_max > 0, the IC is advanced by a random number of
261        solver macro-steps (0..ic_spinup_max) to produce a "spun-up" state.
262        Each call samples its own spinup length independently.
263        """
264        if self.ictype == "gridded":
265            if self._gridded_ics is None:
266                raise RuntimeError("No gridded ICs set. Call set_gridded_ics() first.")
267            
268            # Get the next IC (cycling through)
269            inp_spec = self._gridded_ics[self._gridded_ic_index].clone()
270            self._gridded_ic_index = (self._gridded_ic_index + 1) % len(self._gridded_ics)
271        elif self.ictype == "random":
272            # Use our new configurable method
273            inp_spec = self.random_initial_condition_configurable(mach=self.ic_mach)
274        elif self.ictype == "galewsky":
275            inp_spec = self.solver.galewsky_initial_condition()
276        else:
277            raise ValueError(f"Unknown ictype: {self.ictype}")
278        
279        # IC spinup: advance by random number of solver macro-steps
280        if self.ic_spinup_max > 0:
281            n_spinup = torch.randint(0, self.ic_spinup_max + 1, (1,)).item()
282            for _ in range(n_spinup):
283                inp_spec = self.solver.timestep(inp_spec, self.nsteps)
284        
285        return inp_spec

287    def _get_sample(self):
288        """Override to support gridded IC mode, multi-step trajectories, and multi-history.
289        
290        Returns:
291            inp: Input state(s)
292                - If n_history=1: (C, H, W) - single timestep (backward compatible)
293                - If n_history>1: (n_history*C, H, W) - stacked timesteps [t-(n-1), ..., t-1, t]
294            tar: Target state(s)
295                - If nfuture=0: (C, H, W) - single target
296                - If nfuture>0: (nfuture+1, C, H, W) - trajectory of targets
297        
298        Note: Normalization is handled in __getitem__, not here.
299        """
300        # Note: we always use our own _get_initial_spec path (no delegation to parent)
301        # so that ic_spinup_max is applied consistently to all samples.
302        
303        # Get starting IC in spectral space
304        start_spec = self._get_initial_spec()
305        
306        # For multi-history, we need to generate a trajectory first
307        # Input uses states at [t-(n_history-1), ..., t-1, t], target is at t+1 (or trajectory)
308        
309        if self.n_history > 1:
310            # Generate n_history states by running solver
311            # Start from IC and step forward to build history buffer
312            history_specs = [start_spec.clone()]
313            current_spec = start_spec.clone()
314            for _ in range(self.n_history - 1):
315                current_spec = self.solver.timestep(current_spec, self.nsteps)
316                history_specs.append(current_spec.clone())
317            
318            # Convert history to grid space and stack
319            history_grids = [self.solver.spec2grid(s) for s in history_specs]
320            inp = torch.cat(history_grids, dim=0)  # (n_history*C, H, W)
321            
322            # The "current" state for target generation is the last in history
323            inp_spec_for_target = history_specs[-1]
324        else:
325            # Single history - just use the starting IC
326            inp = self.solver.spec2grid(start_spec)  # (C, H, W)
327            inp_spec_for_target = start_spec
328        
329        # Generate target(s)
330        if self.nfuture > 0:
331            targets = []
332            current_spec = inp_spec_for_target.clone()
333            for _ in range(self.nfuture + 1):
334                current_spec = self.solver.timestep(current_spec, self.nsteps)
335                targets.append(self.solver.spec2grid(current_spec))
336            tar = torch.stack(targets, dim=0)  # (nfuture+1, C, H, W)
337        else:
338            # Single target: one step ahead
339            tar_spec = self.solver.timestep(inp_spec_for_target, self.nsteps)
340            tar = self.solver.spec2grid(tar_spec)  # (C, H, W)
341        
342        return inp, tar

344    def __getitem__(self, index):
345        """Override to handle multi-step trajectory and multi-history normalization.
346        
347        Normalization is applied per-channel (repeating stats for multi-history inputs).
348        """
349        with torch.inference_mode():
350            with torch.no_grad():
351                inp, tar = self._get_sample()
352
353                if self.normalize:
354                    if self.n_history > 1:
355                        # Stack normalization stats for multi-history input
356                        # inp shape: (n_history*C, H, W), stats shape: (C, H, W) or (1, C, 1, 1)
357                        mean_stacked = self.inp_mean.repeat(self.n_history, 1, 1)
358                        var_stacked = self.inp_var.repeat(self.n_history, 1, 1)
359                        inp = (inp - mean_stacked) / torch.sqrt(var_stacked)
360                    else:
361                        inp = (inp - self.inp_mean) / torch.sqrt(self.inp_var)
362                    
363                    # Target normalization (always single C channels per timestep)
364                    tar = (tar - self.inp_mean) / torch.sqrt(self.inp_var)
365
366        return inp.clone(), tar.clone()

288    def _get_sample(self):
289        """Override to support gridded IC mode, multi-step trajectories, and multi-history.
290        
291        Returns:

296                - If nfuture=0: (C, H, W) - single target
297                - If nfuture>0: (nfuture+1, C, H, W) - trajectory of targets
298        
299        Note: Normalization is handled in __getitem__, not here.
300        """
301        # Note: we always use our own _get_initial_spec path (no delegation to parent)
302        # so that ic_spinup_max is applied consistently to all samples.
303        
304        # Get starting IC in spectral space
305        start_spec = self._get_initial_spec()
306        
307        # For multi-history, we need to generate a trajectory first
308        # Input uses states at [t-(n_history-1), ..., t-1, t], target is at t+1 (or trajectory)
309        
310        if self.n_history > 1:

312            # Start from IC and step forward to build history buffer

320            history_grids = [self.solver.spec2grid(s) for s in history_specs]
321            inp = torch.cat(history_grids, dim=0)  # (n_history*C, H, W)
322            
323            # The "current" state for target generation is the last in history
324            inp_spec_for_target = history_specs[-1]
325        else:
326            # Single history - just use the starting IC
327            inp = self.solver.spec2grid(start_spec)  # (C, H, W)
328            inp_spec_for_target = start_spec
329        
330        # Generate target(s)
331        if self.nfuture > 0:
332            targets = []
333            current_spec = inp_spec_for_target.clone()
334            for _ in range(self.nfuture + 1):
335                current_spec = self.solver.timestep(current_spec, self.nsteps)
336                targets.append(self.solver.spec2grid(current_spec))
337            tar = torch.stack(targets, dim=0)  # (nfuture+1, C, H, W)
338        else:

341            tar = self.solver.spec2grid(tar_spec)  # (C, H, W)
342
