DiscoEncodeProcessDecode

DiscoEncodeProcessDecode: DiSCO-based encode -> process -> decode model.

This model replaces the naive patch-based spatial mixing with geodesically-aware DiSCO convolutions. The architecture:

  1. Encode: Physical state -> Latent state (pointwise MLP)
  2. Process: DiSCO spatial mixing + head MLPs + FFN channel mixing
  3. Decode: Latent state -> Physical state (pointwise MLP)

Key differences from EncodeProcessDecode:

  • Uses DiSCO convolutions instead of patch extraction + linear
  • No patching/padding needed (DiSCO handles spherical geometry)
  • theta_cutoff controls receptive field instead of patch_size
  • No FiLM modulation (DiSCO is geodesically aware, sin/cos lat in data)

 32class HeadProcessing(nn.Module):
 33    """
 34    Per-head processing after DiSCO spatial mixing.
 35    Takes DiSCO output features and applies head-specific transformations.
 36    """
 37    def __init__(self, in_dim: int, hidden_dims: List[int], output_dim: int, use_spectral_norm: bool = False):
 38        super().__init__()
 39        layers = []
 40        current_dim = in_dim
 41        for h_dim in hidden_dims:
 42            linear = nn.Linear(current_dim, h_dim)
 43            if use_spectral_norm:
 44                linear = nn.utils.spectral_norm(linear)
 45            layers.append(linear)
 46            layers.append(nn.GELU())
 47            current_dim = h_dim
 48        
 49        last_linear = nn.Linear(current_dim, output_dim)
 50        if use_spectral_norm:
 51            last_linear = nn.utils.spectral_norm(last_linear)
 52        layers.append(last_linear)
 53        self.mlp = nn.Sequential(*layers)
 54
 55    def forward(self, x):
 56        return self.mlp(x)

 59class DiscoMixerBlock(nn.Module):
 60    """
 61    DiSCO-based mixer block with:
 62    1. DiSCO spatial mixing (geodesically-aware convolutions)
 63    2. Per-head MLP processing
 64    3. FFN for channel mixing
 65    
 66    The DiSCO convolution learns shared_hidden kernels that are applied
 67    independently to each channel (depthwise style with shared kernels).
 68    """
 69    def __init__(
 70        self,
 71        model_dim: int,
 72        n_heads: int,
 73        shared_hidden: int,
 74        head_mlp_hidden_dims: List[int],
 75        ffn_hidden_dims: List[int],
 76        img_shape: Tuple[int, int],
 77        kernel_shape: Union[int, Tuple[int, int]] = (5, 5),
 78        theta_cutoff: Optional[float] = None,
 79        basis_type: str = "piecewise linear",
 80        grid: str = "equiangular",
 81        dropout_rate: float = 0.0,
 82        use_residual: bool = True,
 83        use_lat_modulation: bool = False,
 84        use_spectral_norm: bool = False,
 85    ):
 86        super().__init__()
 87        
 88        # model_dim includes sin/cos lat (last 2 channels are immutable)
 89        self.model_dim = model_dim
 90        self.learnable_dim = model_dim - 2
 91        self.head_dim = self.learnable_dim // n_heads
 92        self.n_heads = n_heads
 93        self.shared_hidden = shared_hidden
 94        self.use_residual = use_residual
 95        self.grad_ckpt_level = 0
 96        
 97        if self.learnable_dim % n_heads != 0:
 98            raise ValueError(f"learnable_dim ({self.learnable_dim}) must be divisible by n_heads ({n_heads})")
 99        
100        # DiSCO convolution: learns shared_hidden kernels
101        # Applied to each channel independently via reshape trick
102        # in_channels=1, out_channels=shared_hidden means same kernels for all channels
103        self.disco = DiscreteContinuousConvS2(
104            in_channels=1,
105            out_channels=shared_hidden,
106            in_shape=img_shape,
107            out_shape=img_shape,
108            kernel_shape=kernel_shape,
109            basis_type=basis_type,
110            basis_norm_mode="mean",
111            groups=1,
112            grid_in=grid,
113            grid_out=grid,
114            bias=True,
115            theta_cutoff=theta_cutoff,
116        )
117        
118        # Per-head processing: takes shared_hidden features -> 1 output (for central prediction)
119        # This mirrors the original head_processing structure
120        self.head_processing = nn.ModuleList([
121            HeadProcessing(shared_hidden, head_mlp_hidden_dims, 1, use_spectral_norm=use_spectral_norm)
122            for _ in range(n_heads)
123        ])
124        
125        # FFN for channel mixing (takes full model_dim as input for latitude awareness)
126        ffn_layers = []
127        current_dim = model_dim
128        for h_dim in ffn_hidden_dims:
129            linear = nn.Linear(current_dim, h_dim)
130            if use_spectral_norm:
131                linear = nn.utils.spectral_norm(linear)
132            ffn_layers.append(linear)
133            ffn_layers.append(nn.GELU())
134            current_dim = h_dim
135        
136        last_linear = nn.Linear(current_dim, self.learnable_dim)
137        if use_spectral_norm:
138            last_linear = nn.utils.spectral_norm(last_linear)
139        ffn_layers.append(last_linear)
140        ffn_layers.append(nn.Dropout(dropout_rate))
141        self.ffn = nn.Sequential(*ffn_layers)
142        
143        # Initialize FFN to ignore latitude channels initially
144        with torch.no_grad():
145            self.ffn[0].weight[:, -2:] = 0.0
146        
147        # Optional latitude modulation (FiLM)
148        self.use_lat_modulation = use_lat_modulation
149        if use_lat_modulation:
150            # LatModulator: 2 (sin/cos) -> 2*shared_hidden (gamma, beta)
151            self.lat_modulator = LatModulator(shared_hidden * 2, shared_hidden * 2)
152        else:
153            self.lat_modulator = None
154    
155    def set_grad_ckpt_level(self, level: int) -> None:
156        self.grad_ckpt_level = level
157    
158    def forward(self, x: torch.Tensor) -> torch.Tensor:
159        """
160        Args:
161            x: (B, H, W, model_dim) - latent grid with sin/cos lat as last 2 channels
162        
163        Returns:
164            (B, H, W, model_dim) - processed latent grid
165        """
166        B, H, W, D = x.shape
167        
168        # Separate learnable and immutable channels
169        x_learn = x[..., :-2]  # (B, H, W, learnable_dim)
170        sin_cos = x[..., -2:]  # (B, H, W, 2)
171        
172        # Save for residual
173        x_res = x_learn
174        
175        # --- DiSCO Spatial Mixing ---
176        # Reshape to apply same kernels to each channel: (B, C, H, W) -> (B*C, 1, H, W)
177        x_chw = x_learn.permute(0, 3, 1, 2)  # (B, learnable_dim, H, W)
178        x_flat = x_chw.reshape(B * self.learnable_dim, 1, H, W)
179        
180        # Apply DiSCO: (B*C, 1, H, W) -> (B*C, shared_hidden, H, W)
181        if self.grad_ckpt_level >= 1 and self.training and x_flat.requires_grad:
182            disco_out = checkpoint(self.disco, x_flat, use_reentrant=False)
183        else:
184            disco_out = self.disco(x_flat)
185        
186        # Reshape: (B*C, shared_hidden, H, W) -> (B, C, shared_hidden, H, W)
187        disco_out = disco_out.view(B, self.learnable_dim, self.shared_hidden, H, W)
188        
189        # Permute to (B, H, W, C, shared_hidden) for head processing
190        disco_out = disco_out.permute(0, 3, 4, 1, 2)  # (B, H, W, learnable_dim, shared_hidden)
191        
192        # --- FiLM Latitude Modulation (per-latitude-row) ---
193        if self.use_lat_modulation:
194            # Get sin/cos lat for each row: use first column (same across all columns)
195            sin_cos_per_row = sin_cos[:, :, 0, :]  # (B, H, 2)
196            
197            # Compute gamma/beta: (B, H, 2) -> (B, H, 2*shared_hidden)
198            mod_params = self.lat_modulator(sin_cos_per_row)
199            
200            # Split into gamma/beta: (B, H, shared_hidden) each
201            gamma, beta = torch.chunk(mod_params, 2, dim=-1)
202            
203            # Reshape for broadcasting: (B, H, 1, 1, shared_hidden)
204            gamma = gamma.unsqueeze(2).unsqueeze(3)
205            beta = beta.unsqueeze(2).unsqueeze(3)
206            
207            # Modulate: (B, H, W, C, shared_hidden) * (B, H, 1, 1, shared_hidden)
208            disco_out = disco_out * gamma + beta
209        
210        # --- Per-Head Processing ---
211        # Split by heads: learnable_dim -> n_heads chunks of head_dim
212        head_outputs = []
213        for head_idx in range(self.n_heads):
214            # Extract this head's channels: (B, H, W, head_dim, shared_hidden)
215            start_ch = head_idx * self.head_dim
216            end_ch = start_ch + self.head_dim
217            head_features = disco_out[..., start_ch:end_ch, :]  # (B, H, W, head_dim, shared_hidden)
218            
219            # Flatten spatial for processing: (B*H*W*head_dim, shared_hidden)
220            head_flat = head_features.reshape(-1, self.shared_hidden)
221            
222            # Apply head MLP: -> (B*H*W*head_dim, 1)
223            if self.grad_ckpt_level >= 1 and self.training and head_flat.requires_grad:
224                head_out = checkpoint(self.head_processing[head_idx], head_flat, use_reentrant=False)
225            else:
226                head_out = self.head_processing[head_idx](head_flat)
227            
228            # Reshape back: (B, H, W, head_dim)
229            head_out = head_out.view(B, H, W, self.head_dim)
230            head_outputs.append(head_out)
231        
232        # Concatenate heads: (B, H, W, learnable_dim)
233        x_learn = torch.cat(head_outputs, dim=-1)
234        
235        # First residual connection
236        if self.use_residual:
237            x_learn = x_learn + x_res
238        
239        # --- FFN Channel Mixing ---
240        # Include sin/cos for latitude awareness
241        x_full = torch.cat([x_learn, sin_cos], dim=-1)  # (B, H, W, model_dim)
242        
243        x_res = x_learn
244        x_learn = self.ffn(x_full)  # (B, H, W, learnable_dim)
245        x_learn = x_learn + x_res  # Second residual
246        
247        # Re-append immutable sin/cos
248        return torch.cat([x_learn, sin_cos], dim=-1)

251class DiscoEncodeProcessDecode(nn.Module):
252    """
253    DiSCO-based encode -> process -> decode model.
254    
255    Uses geodesically-aware DiSCO convolutions for spatial mixing,
256    replacing the naive patch-based approach.
257    """
258    
259    def __init__(
260        self,
261        img_size: Tuple[int, int] = (256, 512),
262        in_chans: int = 3,
263        out_chans: int = 3,
264        model_dim: int = 34,
265        n_heads: int = 4,
266        num_mixer_blocks: int = 4,
267        shared_hidden: int = 64,
268        head_mlp_hidden_dims: List[int] = [32],
269        ffn_hidden_dims: List[int] = [512, 256],
270        encoder_hidden_dims: List[int] = [256],
271        kernel_shape: Union[int, Tuple[int, int]] = (5, 5),
272        theta_cutoff: Optional[float] = None,
273        basis_type: str = "piecewise linear",
274        dropout_rate: float = 0.0,
275        grad_ckpt_level: int = 1,
276        grid: str = "equiangular",
277        residual_prediction: bool = True,
278        use_lat_modulation: bool = False,
279        use_spectral_filter: bool = False,
280        spectral_lmax: Optional[int] = None,
281        use_hyperdiffusion: bool = False,
282        hyperdiff_dt: float = 3600.0,
283        hyperdiff_order: int = 4,
284        n_history: int = 1,  # Number of input timesteps (1=single-step, 3=AB3-like)
285        enforce_conservation: bool = False,  # If True, subtract global mean from residual
286        use_spectral_norm: bool = False,
287        **kwargs
288    ):
289        """
290        Args:
291            img_size: (n_lat, n_lon) tuple for grid dimensions
292            in_chans: Number of input physical channels
293            out_chans: Number of output physical channels
294            model_dim: Latent space dimension (includes +2 for sin/cos lat)
295            n_heads: Number of heads for spatial mixing
296            num_mixer_blocks: Number of DiSCO mixer blocks
297            shared_hidden: Number of DiSCO kernels (features per channel)
298            head_mlp_hidden_dims: Hidden dimensions for head MLPs
299            ffn_hidden_dims: Hidden dimensions for channel FFN
300            encoder_hidden_dims: Hidden dimensions for encoder MLP
301            kernel_shape: DiSCO kernel dimensions
302            theta_cutoff: Geodesic cutoff radius (radians), None for default
303            dropout_rate: Dropout rate in FFN
304            grad_ckpt_level: Gradient checkpointing level
305            grid: Grid type ("equiangular" or "legendre-gauss")
306            residual_prediction: If True, model predicts residual
307            use_lat_modulation: If True, apply FiLM modulation to DiSCO features
308            use_spectral_filter: If True, apply spectral lowpass filter at output
309            spectral_lmax: Max spherical harmonic degree for filter (default: nlat//3)
310            use_hyperdiffusion: If True, apply hyperdiffusion in spectral space
311            hyperdiff_dt: Timestep for hyperdiffusion damping (seconds), matches SWE solver
312            hyperdiff_order: Power for hyperdiffusion (default 4 for 4th-order)
313            n_history: Number of input timesteps expected (for multi-history models)
314            enforce_conservation: If True, subtract global mean from residual to guarantee conservation
315            use_spectral_norm: If True, apply spectral normalization to linear layers in mixer blocks
316        """
317        super().__init__()
318        
319        if kwargs:
320            print(f"DiscoEncodeProcessDecode: Ignoring unexpected kwargs: {list(kwargs.keys())}")
321        
322        self.n_lat, self.n_lon = img_size
323        self.in_chans = in_chans
324        self.out_chans = out_chans
325        self.M_features = in_chans + 2  # Physical channels + sin/cos latitude
326        self.model_dim = model_dim
327        self.grid_type = grid
328        self.residual_prediction = residual_prediction
329        self.grad_ckpt_level = grad_ckpt_level
330        self.n_history = n_history  # Store for training script to read
331        
332        if num_mixer_blocks < 1:
333            raise ValueError("num_mixer_blocks must be at least 1.")
334        
335        # --- Encoder and Decoder ---
336        self.encoder = Encoder(self.M_features, model_dim, encoder_hidden_dims)
337        self.decoder = Decoder(model_dim, self.M_features, encoder_hidden_dims[::-1])
338        
339        # --- DiSCO Mixer Blocks ---
340        mixer_blocks = []
341        for block_idx in range(num_mixer_blocks):
342            mixer_blocks.append(
343                DiscoMixerBlock(
344                    model_dim=model_dim,
345                    n_heads=n_heads,
346                    shared_hidden=shared_hidden,
347                    head_mlp_hidden_dims=head_mlp_hidden_dims,
348                    ffn_hidden_dims=ffn_hidden_dims,
349                    img_shape=img_size,
350                    kernel_shape=kernel_shape,
351                    theta_cutoff=theta_cutoff,
352                    basis_type=basis_type,
353                    grid=grid,
354                    dropout_rate=dropout_rate,
355                    use_residual=(block_idx != 0),  # First block no residual
356                    use_lat_modulation=use_lat_modulation,
357                    use_spectral_norm=use_spectral_norm,
358                )
359            )
360        self.mixer = nn.ModuleList(mixer_blocks)
361        
362        # --- Latitude Feature Buffer ---
363        self.register_buffer('lat_features', None, persistent=False)
364        
365        # --- Spectral Filter and Hyperdiffusion (Optional) ---
366        self.use_spectral_filter = use_spectral_filter
367        self.use_hyperdiffusion = use_hyperdiffusion
368        
369        # Set up SHT/ISHT if either spectral filter or hyperdiffusion is enabled
370        if use_spectral_filter or use_hyperdiffusion:
371            lmax = spectral_lmax if spectral_lmax is not None else self.n_lat // 3
372            mmax = lmax  # Isotropic filtering
373            self.spectral_lmax = lmax
374            self.sht = RealSHT(self.n_lat, self.n_lon, lmax=lmax, mmax=mmax, grid=grid)
375            self.isht = InverseRealSHT(self.n_lat, self.n_lon, lmax=lmax, mmax=mmax, grid=grid)
376            
377            if use_spectral_filter:
378                print(f"Spectral filter enabled: lmax={lmax} (hard truncation at l>{lmax})")
379            
380            # Hyperdiffusion setup (matches SWE solver behavior)
381            if use_hyperdiffusion:
382                # Compute Laplacian eigenvalues: -l(l+1) / R^2
383                # For unit sphere (R=1), just use -l(l+1)
384                l = torch.arange(0, lmax).reshape(lmax, 1).float()
385                l = l.expand(lmax, mmax)
386                lap = -l * (l + 1)
387                
388                # Hyperdiffusion: exp(-strength * (lap/lap_max)^order)
389                # Matches SWE solver: exp((-dt/2/3600) * (lap/lap[-1,0])^4)
390                strength = hyperdiff_dt / 2.0 / 3600.0
391                normalized_lap = lap / lap[-1, 0]  # Range [0, 1]
392                hyperdiff = torch.exp(-strength * normalized_lap ** hyperdiff_order)
393                
394                self.register_buffer('hyperdiff', hyperdiff)
395                print(f"Hyperdiffusion enabled: dt={hyperdiff_dt}s, order={hyperdiff_order}, strength={strength:.6f}")
396        else:
397            self.spectral_lmax = None
398            self.sht = None
399            self.isht = None
400            self.register_buffer('hyperdiff', None)
401        
402        # --- Conservation Constraint (Optional) ---
403        self.enforce_conservation = enforce_conservation
404        if enforce_conservation:
405            # Compute quadrature weights for area-weighted global mean
406            # For equiangular grid: weights are cos(lat)
407            if grid == "equiangular":
408                lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat)
409                weights = torch.cos(lat)
410            elif grid == "legendre-gauss":
411                # For Gauss-Legendre, use the quadrature weights
412                _, w = np.polynomial.legendre.leggauss(self.n_lat)
413                weights = torch.tensor(w, dtype=torch.float32)
414            else:
415                lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat)
416                weights = torch.cos(lat)
417            
418            # Normalize and register as buffer: (nlat,)
419            weights = weights / weights.sum()
420            self.register_buffer('conservation_weights', weights)
421            print(f"Conservation constraint enabled: global mean subtracted from residual")
422        else:
423            self.register_buffer('conservation_weights', None)
424        
425        # Set grad checkpoint levels
426        self._set_grad_ckpt_levels(grad_ckpt_level)
427        
428        # Print parameter count
429        self._print_param_counts()

453    def _get_lat_features(self, device, dtype) -> torch.Tensor:
454        """Generate or retrieve cached latitude features."""
455        if self.lat_features is not None:
456            if self.lat_features.device == device and self.lat_features.dtype == dtype:
457                return self.lat_features
458        
459        if self.grid_type == "equiangular":
460            lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat, device=device, dtype=dtype)
461            sin_lat = torch.sin(lat)
462            cos_lat = torch.cos(lat)
463        elif self.grid_type == "legendre-gauss":
464            try:
465                cost, _ = np.polynomial.legendre.leggauss(self.n_lat)
466            except AttributeError:
467                cost = np.linspace(-1, 1, self.n_lat)
468            sin_lat = torch.tensor(cost, device=device, dtype=dtype)
469            cos_lat = torch.sqrt(1 - sin_lat**2)
470        else:
471            lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat, device=device, dtype=dtype)
472            sin_lat = torch.sin(lat)
473            cos_lat = torch.cos(lat)
474        
475        sin_grid = sin_lat.view(-1, 1).expand(self.n_lat, self.n_lon)
476        cos_grid = cos_lat.view(-1, 1).expand(self.n_lat, self.n_lon)
477        features = torch.stack([sin_grid, cos_grid], dim=-1)
478        
479        self.lat_features = features
480        return features

482    def _encode(self, x: torch.Tensor) -> torch.Tensor:
483        """Encode physical state to latent space."""
484        sin_cos_lat = x[..., -2:]
485        encoded_learnable = self.encoder(x)
486        return torch.cat([encoded_learnable, sin_cos_lat], dim=-1)
487    
488    def _process(self, latent_grid: torch.Tensor) -> torch.Tensor:
489        """Apply DiSCO mixer blocks to compute latent delta."""
490        # Apply mixer blocks
491        x = latent_grid
492        for block in self.mixer:
493            if self.grad_ckpt_level >= 2 and self.training and x.requires_grad:
494                x = checkpoint(block, x, use_reentrant=False)
495            else:
496                x = block(x)
497        
498        # Compute delta (difference from input)
499        delta = x - latent_grid
500        
501        # Zero out sin/cos lat delta
502        delta = delta.clone()
503        delta[..., -2:] = 0.0
504        
505        return delta
506    
507    def _decode(self, z: torch.Tensor) -> torch.Tensor:
508        """Decode latent state back to physical space."""
509        return self.decoder(z)

511    def _spectral_filter(self, x: torch.Tensor) -> torch.Tensor:
512        """Apply spectral lowpass filter and/or hyperdiffusion to output.
513        
514        - If use_spectral_filter: Hard truncation at l > lmax (via SHT/ISHT)
515        - If use_hyperdiffusion: Smooth exponential damping in spectral space
516        - Both can be enabled independently or together
517        
518        Input/output: (B, C, H, W)
519        """
520        if self.sht is None or self.isht is None:
521            return x
522        
523        B, C, H, W = x.shape
524        filtered_channels = []
525        for c in range(C):
526            # SHT expects (B, H, W), returns spectral coefficients (B, lmax, mmax)
527            spec = self.sht(x[:, c])
528            
529            # Apply hyperdiffusion in spectral space (element-wise multiply)
530            if self.use_hyperdiffusion and self.hyperdiff is not None:
531                spec = spec * self.hyperdiff
532            
533            # ISHT back to grid space (hard truncation happens automatically via lmax)
534            filtered_channels.append(self.isht(spec))
535        
536        return torch.stack(filtered_channels, dim=1)
537    
538    def forward(self, x: torch.Tensor) -> torch.Tensor:
539        """
540        Forward pass: encode -> process -> decode.
541        
542        Args:
543            x: Input tensor of shape (B, C, H, W) where C = n_history * physical_chans
544        
545        Returns:
546            Prediction tensor of shape (B, out_chans, H, W) - next single timestep
547        """
548        B, C, H, W = x.shape
549        
550        if H != self.n_lat or W != self.n_lon:
551            raise ValueError(
552                f"Input spatial dims ({H}, {W}) don't match model ({self.n_lat}, {self.n_lon})"
553            )
554        
555        # For multi-history: extract most recent state for residual connection
556        # Input is [oldest, ..., newest] stacked along channels
557        if self.n_history > 1:
558            physical_chans = C // self.n_history  # 3 for SWE
559            x_recent = x[:, -physical_chans:, :, :]  # Last timestep (most recent)
560        else:
561            x_recent = x  # Single timestep input
562        
563        # 1. Prepare input: (B, C, H, W) -> (B, H, W, C+2)
564        x_perm = x.permute(0, 2, 3, 1)
565        lat_feats = self._get_lat_features(x.device, x.dtype)
566        lat_feats = lat_feats.unsqueeze(0).expand(B, -1, -1, -1)
567        x_in = torch.cat([x_perm, lat_feats], dim=-1)
568        
569        # 2. Encode
570        latent = self._encode(x_in)
571        
572        # 3. Process (compute delta)
573        delta = self._process(latent)
574        
575        # 4. Apply delta
576        latent_next = latent + delta
577        
578        # 5. Decode
579        decoded = self._decode(latent_next)
580        
581        # 6. Remove latitude features and permute back
582        # For multi-history, take only out_chans (3) from decoded output
583        if self.n_history > 1:
584            out_phys = decoded[..., :self.out_chans]  # Take first out_chans features
585        else:
586            out_phys = decoded[..., :-2]  # Original behavior: remove sin/cos lat
587        out = out_phys.permute(0, 3, 1, 2)
588        
589        # 7. Optional residual in physical space (use most recent timestep)
590        if self.residual_prediction:
591            # If conservation enabled, subtract area-weighted global mean from residual
592            if self.enforce_conservation and self.conservation_weights is not None:
593                # out is the raw residual (B, C, H, W) before adding to input
594                # Compute area-weighted mean for each channel
595                # weights: (nlat,) -> (1, 1, nlat, 1) for broadcasting
596                w = self.conservation_weights.view(1, 1, -1, 1)
597                # Weighted mean: sum over lat/lon, weighted by cos(lat)
598                # (B, C, H, W) * (1, 1, H, 1) -> sum over W, weighted sum over H
599                out_mean = (out * w).sum(dim=(2, 3), keepdim=True) / self.n_lon  # (B, C, 1, 1)
600                out = out - out_mean  # Zero-mean residual
601            out = out + x_recent
602        
603        # 8. Optional spectral lowpass filter and/or hyperdiffusion
604        if self.use_spectral_filter or self.use_hyperdiffusion:
605            out = self._spectral_filter(out)
606        
607        return out

539    def forward(self, x: torch.Tensor) -> torch.Tensor:
540        """

542        
543        Args:
544            x: Input tensor of shape (B, C, H, W) where C = n_history * physical_chans
545        
546        Returns:
547            Prediction tensor of shape (B, out_chans, H, W) - next single timestep
548        """
549        B, C, H, W = x.shape
550        
551        if H != self.n_lat or W != self.n_lon:
552            raise ValueError(
553                f"Input spatial dims ({H}, {W}) don't match model ({self.n_lat}, {self.n_lon})"
554            )
555        
556        # For multi-history: extract most recent state for residual connection
557        # Input is [oldest, ..., newest] stacked along channels
558        if self.n_history > 1:
559            physical_chans = C // self.n_history  # 3 for SWE
560            x_recent = x[:, -physical_chans:, :, :]  # Last timestep (most recent)
561        else:
562            x_recent = x  # Single timestep input
563        
564        # 1. Prepare input: (B, C, H, W) -> (B, H, W, C+2)
565        x_perm = x.permute(0, 2, 3, 1)
566        lat_feats = self._get_lat_features(x.device, x.dtype)
567        lat_feats = lat_feats.unsqueeze(0).expand(B, -1, -1, -1)
568        x_in = torch.cat([x_perm, lat_feats], dim=-1)
569        
570        # 2. Encode
571        latent = self._encode(x_in)
572        
573        # 3. Process (compute delta)
574        delta = self._process(latent)
575        
576        # 4. Apply delta
577        latent_next = latent + delta
578        
579        # 5. Decode
580        decoded = self._decode(latent_next)
581        
582        # 6. Remove latitude features and permute back
583        # For multi-history, take only out_chans (3) from decoded output
584        if self.n_history > 1:
585            out_phys = decoded[..., :self.out_chans]  # Take first out_chans features
586        else:
587            out_phys = decoded[..., :-2]  # Original behavior: remove sin/cos lat
588        out = out_phys.permute(0, 3, 1, 2)
589        
590        # 7. Optional residual in physical space (use most recent timestep)
591        if self.residual_prediction:
592            # If conservation enabled, subtract area-weighted global mean from residual
593            if self.enforce_conservation and self.conservation_weights is not None:
594                # out is the raw residual (B, C, H, W) before adding to input
595                # Compute area-weighted mean for each channel
596                # weights: (nlat,) -> (1, 1, nlat, 1) for broadcasting
597                w = self.conservation_weights.view(1, 1, -1, 1)
598                # Weighted mean: sum over lat/lon, weighted by cos(lat)
599                # (B, C, H, W) * (1, 1, H, 1) -> sum over W, weighted sum over H
600                out_mean = (out * w).sum(dim=(2, 3), keepdim=True) / self.n_lon  # (B, C, 1, 1)
601                out = out - out_mean  # Zero-mean residual
602            out = out + x_recent
603        
604        # 8. Optional spectral lowpass filter and/or hyperdiffusion
605        if self.use_spectral_filter or self.use_hyperdiffusion:
606            out = self._spectral_filter(out)
607        
608        return out
