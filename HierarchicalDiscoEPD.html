HierarchicalDiscoEPD

HierarchicalDiscoEPD: Hierarchical DiSCO encode -> process -> decode model.

This model uses stacked DiSCO convolutions to extract multi-scale spatial features:

  • Level 1: Small FOV (~3x3 at equator) - local/CFL-scale features
  • Level 2: Applied to Level 1 output - medium effective FOV
  • Level 3: Applied to Level 2 output - larger effective FOV
  • etc.

Key design choices:

  • NO activation between DiSCO levels (preserves signed features for gradients)
  • Kernel budget split across levels (not multiplied)
  • Concatenate all levels' features for Head MLPs
  • FFN for channel mixing as before

 31class HierarchicalDiscoExtractor(nn.Module):
 32    """
 33    Extracts multi-scale spatial features using stacked DiSCO convolutions.
 34    
 35    Each level applies DiSCO to the previous level's output (no activation between).
 36    All levels' features are concatenated for downstream processing.
 37    """
 38    
 39    def __init__(
 40        self,
 41        num_levels: int,
 42        kernels_per_level: Union[int, List[int]],
 43        img_shape: Tuple[int, int],
 44        theta_cutoff: Union[float, List[float]],
 45        kernel_shape: Union[int, Tuple[int, int], List[Tuple[int, int]]] = (3, 3),
 46        basis_type: str = "piecewise linear",
 47        grid: str = "equiangular",
 48    ):
 49        """
 50        Args:
 51            num_levels: Number of hierarchical levels
 52            kernels_per_level: Kernels per level (int for equal, list for custom)
 53            img_shape: (n_lat, n_lon)
 54            theta_cutoff: Geodesic cutoff (float for same across levels, list for per-level)
 55            kernel_shape: DiSCO kernel shape (single for all levels, list for per-level)
 56            basis_type: DiSCO basis type
 57            grid: Grid type
 58        """
 59        super().__init__()
 60        
 61        self.num_levels = num_levels
 62        
 63        # Handle kernels_per_level as int or list
 64        if isinstance(kernels_per_level, int):
 65            self.kernels_per_level = [kernels_per_level] * num_levels
 66        else:
 67            assert len(kernels_per_level) == num_levels
 68            self.kernels_per_level = list(kernels_per_level)
 69        
 70        # Handle theta_cutoff as float or list
 71        if isinstance(theta_cutoff, (int, float)):
 72            self.theta_cutoffs = [float(theta_cutoff)] * num_levels
 73        else:
 74            assert len(theta_cutoff) == num_levels, f"theta_cutoff list length ({len(theta_cutoff)}) must match num_levels ({num_levels})"
 75            self.theta_cutoffs = list(theta_cutoff)
 76        
 77        # Handle kernel_shape as single value or list
 78        if isinstance(kernel_shape, (int, tuple)):
 79            self.kernel_shapes = [kernel_shape] * num_levels
 80        else:
 81            assert len(kernel_shape) == num_levels, f"kernel_shape list length ({len(kernel_shape)}) must match num_levels ({num_levels})"
 82            self.kernel_shapes = list(kernel_shape)
 83        
 84        self.total_kernels = sum(self.kernels_per_level)
 85        
 86        # Create DiSCO layers for each level
 87        # Level 1: in_channels=1 (applied per channel)
 88        # Level 2+: in_channels=kernels from previous level
 89        self.disco_layers = nn.ModuleList()
 90        
 91        for level_idx in range(num_levels):
 92            if level_idx == 0:
 93                in_ch = 1  # First level: applied to each channel independently
 94            else:
 95                in_ch = self.kernels_per_level[level_idx - 1]  # Previous level's output
 96            
 97            out_ch = self.kernels_per_level[level_idx]
 98            
 99            self.disco_layers.append(
100                DiscreteContinuousConvS2(
101                    in_channels=in_ch,
102                    out_channels=out_ch,
103                    in_shape=img_shape,
104                    out_shape=img_shape,
105                    kernel_shape=self.kernel_shapes[level_idx],
106                    basis_type=basis_type,
107                    basis_norm_mode="mean",
108                    groups=1,
109                    grid_in=grid,
110                    grid_out=grid,
111                    bias=True,
112                    theta_cutoff=self.theta_cutoffs[level_idx],
113                )
114            )
115    
116    def forward(self, x: torch.Tensor) -> torch.Tensor:
117        """
118        Args:
119            x: (B*C, 1, H, W) - single channel per batch element
120        
121        Returns:
122            (B*C, total_kernels, H, W) - concatenated multi-scale features
123        """
124        level_outputs = []
125        current = x  # (B*C, 1, H, W)
126        
127        for level_idx, disco in enumerate(self.disco_layers):
128            # Apply DiSCO (no activation between levels!)
129            current = disco(current)  # (B*C, kernels_this_level, H, W)
130            
131            # Store this level's output
132            level_outputs.append(current)
133        
134        # Concatenate all levels: (B*C, total_kernels, H, W)
135        return torch.cat(level_outputs, dim=1)

138class HierarchicalDiscoMixerBlock(nn.Module):
139    """
140    Mixer block using hierarchical DiSCO for multi-scale spatial feature extraction.
141    """
142    
143    def __init__(
144        self,
145        model_dim: int,
146        n_heads: int,
147        num_levels: int,
148        kernels_per_level: Union[int, List[int]],
149        head_mlp_hidden_dims: List[int],
150        ffn_hidden_dims: List[int],
151        img_shape: Tuple[int, int],
152        theta_cutoff: Union[float, List[float]],
153        kernel_shape: Union[int, Tuple[int, int], List[Tuple[int, int]]] = (3, 3),
154        basis_type: str = "piecewise linear",
155        grid: str = "equiangular",
156        dropout_rate: float = 0.0,
157        use_residual: bool = True,
158        use_lat_modulation: bool = False,
159    ):
160        super().__init__()
161        
162        self.model_dim = model_dim
163        self.learnable_dim = model_dim - 2  # Exclude sin/cos
164        self.head_dim = self.learnable_dim // n_heads
165        self.n_heads = n_heads
166        self.use_residual = use_residual
167        self.grad_ckpt_level = 0
168        
169        if self.learnable_dim % n_heads != 0:
170            raise ValueError(f"learnable_dim ({self.learnable_dim}) must be divisible by n_heads ({n_heads})")
171        
172        # Hierarchical DiSCO extractor
173        self.disco_extractor = HierarchicalDiscoExtractor(
174            num_levels=num_levels,
175            kernels_per_level=kernels_per_level,
176            img_shape=img_shape,
177            theta_cutoff=theta_cutoff,
178            kernel_shape=kernel_shape,
179            basis_type=basis_type,
180            grid=grid,
181        )
182        
183        total_kernels = self.disco_extractor.total_kernels
184        
185        # Per-head processing: takes total_kernels features -> 1 output
186        self.head_processing = nn.ModuleList([
187            self._build_head_mlp(total_kernels, head_mlp_hidden_dims)
188            for _ in range(n_heads)
189        ])
190        
191        # FFN for channel mixing
192        ffn_layers = []
193        current_dim = model_dim
194        for h_dim in ffn_hidden_dims:
195            ffn_layers.append(nn.Linear(current_dim, h_dim))
196            ffn_layers.append(nn.GELU())
197            current_dim = h_dim
198        ffn_layers.append(nn.Linear(current_dim, self.learnable_dim))
199        ffn_layers.append(nn.Dropout(dropout_rate))
200        self.ffn = nn.Sequential(*ffn_layers)
201        
202        # Initialize FFN to ignore latitude channels initially
203        with torch.no_grad():
204            self.ffn[0].weight[:, -2:] = 0.0
205        
206        # Optional latitude modulation (FiLM)
207        self.use_lat_modulation = use_lat_modulation
208        if use_lat_modulation:
209            # LatModulator: 2 (sin/cos) -> 2*total_kernels (gamma, beta)
210            self.lat_modulator = LatModulator(total_kernels * 2, total_kernels * 2)
211        else:
212            self.lat_modulator = None
213    
214    def _build_head_mlp(self, in_dim: int, hidden_dims: List[int]) -> nn.Module:
215        """Build MLP for head processing."""
216        layers = []
217        current_dim = in_dim
218        for h_dim in hidden_dims:
219            layers.append(nn.Linear(current_dim, h_dim))
220            layers.append(nn.GELU())
221            current_dim = h_dim
222        layers.append(nn.Linear(current_dim, 1))
223        return nn.Sequential(*layers)
224    
225    def set_grad_ckpt_level(self, level: int) -> None:
226        self.grad_ckpt_level = level
227    
228    def forward(self, x: torch.Tensor) -> torch.Tensor:
229        """
230        Args:
231            x: (B, H, W, model_dim)
232        
233        Returns:
234            (B, H, W, model_dim)
235        """
236        B, H, W, D = x.shape
237        
238        # Separate learnable and immutable channels
239        x_learn = x[..., :-2]  # (B, H, W, learnable_dim)
240        sin_cos = x[..., -2:]  # (B, H, W, 2)
241        
242        x_res = x_learn
243        
244        # --- Hierarchical DiSCO Feature Extraction ---
245        # Reshape: (B, H, W, C) -> (B, C, H, W) -> (B*C, 1, H, W)
246        x_chw = x_learn.permute(0, 3, 1, 2)  # (B, C, H, W)
247        x_flat = x_chw.reshape(B * self.learnable_dim, 1, H, W)
248        
249        # Extract hierarchical features
250        if self.grad_ckpt_level >= 1 and self.training and x_flat.requires_grad:
251            disco_out = checkpoint(self.disco_extractor, x_flat, use_reentrant=False)
252        else:
253            disco_out = self.disco_extractor(x_flat)  # (B*C, total_kernels, H, W)
254        
255        # Reshape: (B*C, K, H, W) -> (B, C, K, H, W) -> (B, H, W, C, K)
256        total_kernels = disco_out.shape[1]
257        disco_out = disco_out.view(B, self.learnable_dim, total_kernels, H, W)
258        disco_out = disco_out.permute(0, 3, 4, 1, 2)  # (B, H, W, C, K)
259        
260        # --- FiLM Latitude Modulation (per-latitude-row) ---
261        if self.use_lat_modulation and self.lat_modulator is not None:
262            # Get sin/cos lat for each row: use first column (same across all columns)
263            sin_cos_per_row = sin_cos[:, :, 0, :]  # (B, H, 2)
264            
265            # Compute gamma/beta: (B, H, 2) -> (B, H, 2*total_kernels)
266            mod_params = self.lat_modulator(sin_cos_per_row)
267            
268            # Split into gamma/beta: (B, H, total_kernels) each
269            gamma, beta = torch.chunk(mod_params, 2, dim=-1)
270            
271            # Reshape for broadcasting: (B, H, 1, 1, total_kernels)
272            gamma = gamma.unsqueeze(2).unsqueeze(3)
273            beta = beta.unsqueeze(2).unsqueeze(3)
274            
275            # Modulate: (B, H, W, C, K) * (B, H, 1, 1, K)
276            disco_out = disco_out * gamma + beta
277        
278        # --- Per-Head Processing ---
279        head_outputs = []
280        for head_idx in range(self.n_heads):
281            start_ch = head_idx * self.head_dim
282            end_ch = start_ch + self.head_dim
283            head_features = disco_out[..., start_ch:end_ch, :]  # (B, H, W, head_dim, K)
284            
285            # Flatten for MLP: (B*H*W*head_dim, K)
286            head_flat = head_features.reshape(-1, total_kernels)
287            
288            # Apply head MLP
289            if self.grad_ckpt_level >= 1 and self.training and head_flat.requires_grad:
290                head_out = checkpoint(self.head_processing[head_idx], head_flat, use_reentrant=False)
291            else:
292                head_out = self.head_processing[head_idx](head_flat)  # (B*H*W*head_dim, 1)
293            
294            # Reshape: (B, H, W, head_dim)
295            head_out = head_out.view(B, H, W, self.head_dim)
296            head_outputs.append(head_out)
297        
298        x_learn = torch.cat(head_outputs, dim=-1)  # (B, H, W, learnable_dim)
299        
300        # First residual
301        if self.use_residual:
302            x_learn = x_learn + x_res
303        
304        # --- FFN Channel Mixing ---
305        x_full = torch.cat([x_learn, sin_cos], dim=-1)
306        x_res = x_learn
307        x_learn = self.ffn(x_full)
308        x_learn = x_learn + x_res  # Second residual
309        
310        return torch.cat([x_learn, sin_cos], dim=-1)

321        self,

329        kernels_per_level: Union[int, List[int]] = 20,
330        head_mlp_hidden_dims: List[int] = [32],
331        ffn_hidden_dims: List[int] = [512, 256],
332        encoder_hidden_dims: List[int] = [128],
333        kernel_shape: Union[int, Tuple[int, int], List[Tuple[int, int]]] = (3, 3),
334        theta_cutoff: Optional[Union[float, List[float]]] = None,
335        basis_type: str = "piecewise linear",
336        dropout_rate: float = 0.0,
337        grad_ckpt_level: int = 1,
338        grid: str = "equiangular",
339        residual_prediction: bool = True,
340        use_lat_modulation: bool = False,
341        use_spectral_filter: bool = False,
342        spectral_lmax: Optional[int] = None,
343        **kwargs
344    ):
345        super().__init__()
346        
347        if kwargs:
348            print(f"HierarchicalDiscoEPD: Ignoring unexpected kwargs: {list(kwargs.keys())}")
349        
350        self.n_lat, self.n_lon = img_size
351        self.in_chans = in_chans
352        self.out_chans = out_chans
353        self.M_features = in_chans + 2
354        self.model_dim = model_dim
355        self.grid_type = grid
356        self.residual_prediction = residual_prediction
357        self.grad_ckpt_level = grad_ckpt_level
358        
359        # Default theta_cutoff: ~3x3 at equator
360        if theta_cutoff is None:
361            theta_cutoff = (3 * torch.pi) / (torch.pi**0.5 * img_size[0])
362        
363        # --- Encoder and Decoder ---
364        self.encoder = Encoder(self.M_features, model_dim, encoder_hidden_dims)
365        self.decoder = Decoder(model_dim, self.M_features, encoder_hidden_dims[::-1])
366        
367        # --- Hierarchical DiSCO Mixer Blocks ---
368        mixer_blocks = []
369        for block_idx in range(num_mixer_blocks):
370            mixer_blocks.append(
371                HierarchicalDiscoMixerBlock(
372                    model_dim=model_dim,
373                    n_heads=n_heads,
374                    num_levels=num_levels,
375                    kernels_per_level=kernels_per_level,
376                    head_mlp_hidden_dims=head_mlp_hidden_dims,
377                    ffn_hidden_dims=ffn_hidden_dims,
378                    img_shape=img_size,

380                    kernel_shape=kernel_shape,

382                    grid=grid,
383                    dropout_rate=dropout_rate,
384                    use_residual=(block_idx != 0),
385                    use_lat_modulation=use_lat_modulation,
386                )
387            )
388        self.mixer = nn.ModuleList(mixer_blocks)
389        
390        # --- Latitude Feature Buffer ---
391        self.register_buffer('lat_features', None, persistent=False)
392        
393        # --- Spectral Filter (Optional) ---
394        self.use_spectral_filter = use_spectral_filter
395        if use_spectral_filter:
396            lmax = spectral_lmax if spectral_lmax is not None else self.n_lat // 3
397            mmax = lmax  # Isotropic filtering
398            self.spectral_lmax = lmax
399            self.sht = RealSHT(self.n_lat, self.n_lon, lmax=lmax, mmax=mmax, grid=grid)
400            self.isht = InverseRealSHT(self.n_lat, self.n_lon, lmax=lmax, mmax=mmax, grid=grid)
401            print(f"Spectral filter enabled: lmax={lmax} (removes modes > {lmax})")
402        else:
403            self.spectral_lmax = None
404            self.sht = None
405            self.isht = None
406        
407        # Set grad checkpoint levels
408        self._set_grad_ckpt_levels(grad_ckpt_level)
409        
410        # Print parameter count
411        self._print_param_counts()

433    def _get_lat_features(self, device, dtype) -> torch.Tensor:
434        if self.lat_features is not None:
435            if self.lat_features.device == device and self.lat_features.dtype == dtype:
436                return self.lat_features
437        
438        if self.grid_type == "equiangular":
439            lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat, device=device, dtype=dtype)
440            sin_lat = torch.sin(lat)
441            cos_lat = torch.cos(lat)
442        elif self.grid_type == "legendre-gauss":
443            try:
444                cost, _ = np.polynomial.legendre.leggauss(self.n_lat)
445            except AttributeError:
446                cost = np.linspace(-1, 1, self.n_lat)
447            sin_lat = torch.tensor(cost, device=device, dtype=dtype)
448            cos_lat = torch.sqrt(1 - sin_lat**2)
449        else:
450            lat = torch.linspace(-np.pi/2, np.pi/2, self.n_lat, device=device, dtype=dtype)
451            sin_lat = torch.sin(lat)
452            cos_lat = torch.cos(lat)
453        
454        sin_grid = sin_lat.view(-1, 1).expand(self.n_lat, self.n_lon)
455        cos_grid = cos_lat.view(-1, 1).expand(self.n_lat, self.n_lon)
456        features = torch.stack([sin_grid, cos_grid], dim=-1)
457        
458        self.lat_features = features
459        return features

461    def _encode(self, x: torch.Tensor) -> torch.Tensor:
462        sin_cos_lat = x[..., -2:]
463        encoded_learnable = self.encoder(x)
464        return torch.cat([encoded_learnable, sin_cos_lat], dim=-1)
465    
466    def _process(self, latent_grid: torch.Tensor) -> torch.Tensor:
467        x = latent_grid
468        for block in self.mixer:
469            if self.grad_ckpt_level >= 2 and self.training and x.requires_grad:
470                x = checkpoint(block, x, use_reentrant=False)
471            else:
472                x = block(x)
473        
474        delta = x - latent_grid
475        delta = delta.clone()
476        delta[..., -2:] = 0.0
477        return delta
478    
479    def _decode(self, z: torch.Tensor) -> torch.Tensor:
480        return self.decoder(z)
481    
482    def _spectral_filter(self, x: torch.Tensor) -> torch.Tensor:
483        """Apply spectral lowpass filter to output.
484        
485        Removes high-frequency spherical harmonic modes (l > lmax).
486        Input/output: (B, C, H, W)
487        """
488        if self.sht is None or self.isht is None:
489            return x
490        
491        B, C, H, W = x.shape
492        filtered_channels = []
493        for c in range(C):
494            # SHT expects (B, H, W), returns spectral coefficients
495            spec = self.sht(x[:, c])
496            # ISHT back to grid space (truncation happens automatically via lmax)
497            filtered_channels.append(self.isht(spec))
498        
499        return torch.stack(filtered_channels, dim=1)
500    
501    def forward(self, x: torch.Tensor) -> torch.Tensor:
502        B, C, H, W = x.shape
503        
504        if H != self.n_lat or W != self.n_lon:
505            raise ValueError(f"Input dims ({H}, {W}) don't match model ({self.n_lat}, {self.n_lon})")
506        
507        # Prepare input
508        x_perm = x.permute(0, 2, 3, 1)
509        lat_feats = self._get_lat_features(x.device, x.dtype)
510        lat_feats = lat_feats.unsqueeze(0).expand(B, -1, -1, -1)
511        x_in = torch.cat([x_perm, lat_feats], dim=-1)
512        
513        # Encode
514        latent = self._encode(x_in)
515        
516        # Process
517        delta = self._process(latent)
518        latent_next = latent + delta
519        
520        # Decode
521        decoded = self._decode(latent_next)
522        
523        # Output
524        out_phys = decoded[..., :-2]
525        out = out_phys.permute(0, 3, 1, 2)
526        
527        if self.residual_prediction:
528            out = out + x
529        
530        # Optional spectral lowpass filter
531        if self.use_spectral_filter:
532            out = self._spectral_filter(out)
533        
534        return out

291                head_out = checkpoint(self.head_processing[head_idx], head_flat, use_reentrant=False)

294

296            head_out = head_out.view(B, H, W, self.head_dim)
297            head_outputs.append(head_out)
298        
299        x_learn = torch.cat(head_outputs, dim=-1)  # (B, H, W, learnable_dim)
300        
301        # First residual
302        if self.use_residual:
303            x_learn = x_learn + x_res
304        
305        # --- FFN Channel Mixing ---
306        x_full = torch.cat([x_learn, sin_cos], dim=-1)
307        x_res = x_learn
308        x_learn = self.ffn(x_full)
309        x_learn = x_learn + x_res  # Second residual
310        
311        return torch.cat([x_learn, sin_cos], dim=-1)
